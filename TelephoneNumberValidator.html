Jake's Telephone Number Validator

My Code Solution:

function telephoneCheck(str) {
    // Regex to match all num possibilities
    let regex = /^(1\s?)?(\(\d{3}\)|\d{3})[\s\-]?\d{3}[\s\-]?\d{4}$/;

    // Test str matches
    let isValid = regex.test(str);

    // Return true/false
    return isValid;
}

  • Test: 555-555-5555 | Expected Output: True
  • Test: 1 456 789 4444 | Expected Output: True
  • Test: 2 757 622-7382 | Expected Output: False
